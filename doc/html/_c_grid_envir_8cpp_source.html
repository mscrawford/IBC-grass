554 void CGridEnvir::SeedRain(){
555 
556  string PFT_ID, PFTtype, Cltype;
557  //size_t posc;
558  SPftTraits *pfttraits;
559 // SclonalTraits *cltraits;
560  double nseeds =0;
561 
562  double SeedFracClonal = 0.1; //ratio between seed input for clonal and non-clonal seeds;
563 
564  //for all plants..
565  for (map<string, long>::const_iterator it = PftInitList.begin();
566  it!= PftInitList.end(); ++it){
567 
568  PFT_ID = it->first;
569 
570 // cltraits=getClLink(PFT_ID);
571  pfttraits=SPftTraits::getPftLink(PFT_ID);
572 
573  switch(SRunPara::RunPara.SeedRainType){
574  // [SeedInput] == seed NUMBER & equal seed NUMBER for each PFT
575  case 1: nseeds = SRunPara::RunPara.SeedInput/(CEnvir::SeedRainGr.NPftSeedSize[0] +
576  CEnvir::SeedRainGr.NPftSeedSize[1] + CEnvir::SeedRainGr.NPftSeedSize[2]);
577  break;

579  case 2: nseeds = SRunPara::RunPara.SeedInput/(CEnvir::SeedRainGr.NPftSeedSize[0]*0.1 +
580  CEnvir::SeedRainGr.NPftSeedSize[1]*0.3 + CEnvir::SeedRainGr.NPftSeedSize[2]*1.0);
581  break;
582  // [SeedInput] == seed MASS & equal seed NUMBER for each PFT
583  case 3: nseeds = SRunPara::RunPara.SeedInput/(CEnvir::SeedRainGr.NPftSeedSize[0] +
584  CEnvir::SeedRainGr.NPftSeedSize[1]/3.0 + CEnvir::SeedRainGr.NPftSeedSize[2]/10.0)*0.1/pfttraits->SeedMass;
585  break;
586  // [SeedInput] == seed MASS& equal seed MASS for each PFT

588  CEnvir::SeedRainGr.NPftSeedSize[1] + CEnvir::SeedRainGr.NPftSeedSize[2]) / pfttraits->SeedMass;
