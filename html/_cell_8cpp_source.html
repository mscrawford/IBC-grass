233 void CCell::AboveComp()
234 {
235  if (AbovePlantList.empty()) return;
236  if (SRunPara::RunPara.AboveCompMode==asymtot){
237  //total asymmetry only for above plant competition
238  sort(AbovePlantList.begin(),AbovePlantList.end(),CPlant::CompareShoot);
239  CPlant* plant=*AbovePlantList.begin(); //biggest plant
240  plant->Auptake+=AResConc;
241  return;
242  }
243  int symm=1; if (SRunPara::RunPara.AboveCompMode==asympart) symm=2;
244  double comp_tot=0, comp_c=0;
245 
246  //1. sum of resource requirement
247  for (plant_iter iter=AbovePlantList.begin(); iter!=AbovePlantList.end(); ++iter){
248  CPlant* plant=*iter;
249  comp_tot+=plant->comp_coef(1,symm)
250  *prop_res(plant->pft(),1,SRunPara::RunPara.Version);
251  }
252  //2. distribute resources
253  for (plant_iter iter=AbovePlantList.begin(); iter!=AbovePlantList.end(); ++iter){
254  CPlant* plant=*iter;
255  comp_c=plant->comp_coef(1,symm)
256  *prop_res(plant->pft(),1,SRunPara::RunPara.Version);
257  plant->Auptake+=AResConc*comp_c/comp_tot;
258  }
259 }//end above_comp

299 double CCell::prop_res(const string type,const int layer,const int version)const{
300  switch (version) {
301  case 0:
302  return 1;break;
303  case 1:
304  if (layer == 1) {
305  map<string, int>::const_iterator noa = PftNIndA.find(type);
306  if (noa != PftNIndA.end())
307  return 1.0 / sqrt(noa->second);
308  }
309  if (layer == 2) {
310  map<string, int>::const_iterator nob = PftNIndB.find(type);
311  if (nob != PftNIndB.end())
312  return 1.0 / sqrt(nob->second);
313 
314  }
315  break;
316  case 2:
317  if (layer == 1)
318  return NPftA / (1.0 + NPftA);
319  if (layer == 2)
320  return NPftB / (1.0 + NPftB);
321  break;
322  default:
323  cerr << "CCell::prop_res() - wrong input";
324  exit(3);break;
325  }
326  return -1; //should not be reached
327 }//end CCell::prop_res
